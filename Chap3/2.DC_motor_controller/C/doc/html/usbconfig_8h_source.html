170 /* ------------------- Fine Control over USB Descriptors ------------------- */
171 /* If you don't want to use the driver's default USB descriptors, you can
172  * provide our own. These can be provided as (1) fixed length static data in
173  * flash memory, (2) fixed length static data in RAM or (3) dynamically at
174  * runtime in the function usbFunctionDescriptor(). See usbdrv.h for more
175  * information about this function.
176  * Descriptor handling is configured through the descriptor's properties. If
177  * no properties are defined or if they are 0, the default descriptor is used.
178  * Possible properties are:
179  * + USB_PROP_IS_DYNAMIC: The data for the descriptor should be fetched
180  * at runtime via usbFunctionDescriptor().
181  * + USB_PROP_IS_RAM: The data returned by usbFunctionDescriptor() or found
182  * in static memory is in RAM, not in flash memory.
183  * + USB_PROP_LENGTH(len): If the data is in static memory (RAM or flash),
184  * the driver must know the descriptor's length. The descriptor itself is
185  * found at the address of a well known identifier (see below).
186  * List of static descriptor names (must be declared PROGMEM if in flash):
187  * char usbDescriptorDevice[];
188  * char usbDescriptorConfiguration[];
189  * char usbDescriptorHidReport[];
190  * char usbDescriptorString0[];
191  * int usbDescriptorStringVendor[];
192  * int usbDescriptorStringDevice[];
193  * int usbDescriptorStringSerialNumber[];
194  * Other descriptors can't be provided statically, they must be provided
195  * dynamically at runtime.
196  *
197  * Descriptor properties are or-ed or added together, e.g.:
198  * #define USB_CFG_DESCR_PROPS_DEVICE (USB_PROP_IS_RAM | USB_PROP_LENGTH(18))
199  *
200  * The following descriptors are defined:
201  * USB_CFG_DESCR_PROPS_DEVICE
202  * USB_CFG_DESCR_PROPS_CONFIGURATION
203  * USB_CFG_DESCR_PROPS_STRINGS
204  * USB_CFG_DESCR_PROPS_STRING_0
205  * USB_CFG_DESCR_PROPS_STRING_VENDOR
206  * USB_CFG_DESCR_PROPS_STRING_PRODUCT
207  * USB_CFG_DESCR_PROPS_STRING_SERIAL_NUMBER
208  * USB_CFG_DESCR_PROPS_HID
209  * USB_CFG_DESCR_PROPS_HID_REPORT
210  * USB_CFG_DESCR_PROPS_UNKNOWN (for all descriptors not handled by the driver)
211  *
212  */
