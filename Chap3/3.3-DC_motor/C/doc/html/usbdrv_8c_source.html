308 static void usbProcessRx(uchar *data, uchar len)
309 {
310 usbRequest_t *rq = (void *)data;
311 uchar replyLen = 0, flags = USB_FLG_USE_DEFAULT_RW;
312 /* We use if() cascades because the compare is done byte-wise while switch()
313  * is int-based. The if() cascades are therefore more efficient.
314  */
315 /* usbRxToken can be:
316  * 0x2d 00101101 (USBPID_SETUP for endpoint 0)
317  * 0xe1 11100001 (USBPID_OUT for endpoint 0)
318  * 0xff 11111111 (USBPID_OUT for endpoint 1)
319  */
320  DBG2(0x10 + ((usbRxToken >> 1) & 3), data, len); /* SETUP0=12; OUT0=10; OUT1=13 */
321 #if USB_CFG_IMPLEMENT_FN_WRITEOUT
322  if(usbRxToken == 0xff){
323  usbFunctionWriteOut(data, len);
324  return; /* no reply expected, hence no usbMsgPtr, usbMsgFlags, usbMsgLen set */
325  }
326 #endif
327  if(usbRxToken == (uchar)USBPID_SETUP){
328  usbTxLen = USBPID_NAK; /* abort pending transmit */
329  if(len == 8){ /* Setup size must be always 8 bytes. Ignore otherwise. */
330  uchar type = rq->bmRequestType & USBRQ_TYPE_MASK;
331  if(type == USBRQ_TYPE_STANDARD){
332  #define SET_REPLY_LEN(len) replyLen = (len); usbMsgPtr = replyData
333  /* This macro ensures that replyLen and usbMsgPtr are always set in the same way.
334  * That allows optimization of common code in if() branches */
335  uchar *replyData = usbTxBuf + 9; /* there is 3 bytes free space at the end of the buffer */
336  replyData[0] = 0; /* common to USBRQ_GET_STATUS and USBRQ_GET_INTERFACE */
337  if(rq->bRequest == USBRQ_GET_STATUS){ /* 0 */
338  uchar __attribute__((__unused__)) recipient = rq->bmRequestType & USBRQ_RCPT_MASK; /* assign arith ops to variables to enforce byte size */
339 #if USB_CFG_IS_SELF_POWERED
340  if(recipient == USBRQ_RCPT_DEVICE)
341  replyData[0] = USB_CFG_IS_SELF_POWERED;
342 #endif
343 #if USB_CFG_HAVE_INTRIN_ENDPOINT && USB_CFG_IMPLEMENT_HALT
344  if(recipient == USBRQ_RCPT_ENDPOINT && rq->wIndex.bytes[0] == 0x81) /* request status for endpoint 1 */
345  replyData[0] = usbTxLen1 == USBPID_STALL;
346 #endif
347  replyData[1] = 0;
348  SET_REPLY_LEN(2);
349  }else if(rq->bRequest == USBRQ_SET_ADDRESS){ /* 5 */
350  usbNewDeviceAddr = rq->wValue.bytes[0];
351  }else if(rq->bRequest == USBRQ_GET_DESCRIPTOR){ /* 6 */
352  flags = USB_FLG_MSGPTR_IS_ROM | USB_FLG_USE_DEFAULT_RW;
353  if(rq->wValue.bytes[1] == USBDESCR_DEVICE){ /* 1 */
354  GET_DESCRIPTOR(USB_CFG_DESCR_PROPS_DEVICE, usbDescriptorDevice)
355  }else if(rq->wValue.bytes[1] == USBDESCR_CONFIG){ /* 2 */
356  GET_DESCRIPTOR(USB_CFG_DESCR_PROPS_CONFIGURATION, usbDescriptorConfiguration)
357  }else if(rq->wValue.bytes[1] == USBDESCR_STRING){ /* 3 */
358 #if USB_CFG_DESCR_PROPS_STRINGS & USB_PROP_IS_DYNAMIC
359  if(USB_CFG_DESCR_PROPS_STRINGS & USB_PROP_IS_RAM)
360  flags &= ~USB_FLG_MSGPTR_IS_ROM;
361  replyLen = usbFunctionDescriptor(rq);
362 #else /* USB_CFG_DESCR_PROPS_STRINGS & USB_PROP_IS_DYNAMIC */
363  if(rq->wValue.bytes[0] == 0){ /* descriptor index */
364  GET_DESCRIPTOR(USB_CFG_DESCR_PROPS_STRING_0, usbDescriptorString0)
365  }else if(rq->wValue.bytes[0] == 1){
366  GET_DESCRIPTOR(USB_CFG_DESCR_PROPS_STRING_VENDOR, usbDescriptorStringVendor)
367  }else if(rq->wValue.bytes[0] == 2){
368  GET_DESCRIPTOR(USB_CFG_DESCR_PROPS_STRING_PRODUCT, usbDescriptorStringDevice)
369  }else if(rq->wValue.bytes[0] == 3){
370  GET_DESCRIPTOR(USB_CFG_DESCR_PROPS_STRING_SERIAL_NUMBER, usbDescriptorStringSerialNumber)
371  }else if(USB_CFG_DESCR_PROPS_UNKNOWN & USB_PROP_IS_DYNAMIC){
372  replyLen = usbFunctionDescriptor(rq);
373  }
374 #endif /* USB_CFG_DESCR_PROPS_STRINGS & USB_PROP_IS_DYNAMIC */
375  }else if(rq->wValue.bytes[1] == USBDESCR_HID){ /* 0x21 */
376  GET_DESCRIPTOR(USB_CFG_DESCR_PROPS_HID, usbDescriptorConfiguration + 18)
377  }else if(rq->wValue.bytes[1] == USBDESCR_HID_REPORT){ /* 0x22 */
378  GET_DESCRIPTOR(USB_CFG_DESCR_PROPS_HID_REPORT, usbDescriptorHidReport)
379  }else if(USB_CFG_DESCR_PROPS_UNKNOWN & USB_PROP_IS_DYNAMIC){
380  replyLen = usbFunctionDescriptor(rq);
381  }
382  }else if(rq->bRequest == USBRQ_GET_CONFIGURATION){ /* 8 */
383  replyData = &usbConfiguration; /* send current configuration value */
384  SET_REPLY_LEN(1);
385  }else if(rq->bRequest == USBRQ_SET_CONFIGURATION){ /* 9 */
386  usbConfiguration = rq->wValue.bytes[0];
387 #if USB_CFG_IMPLEMENT_HALT
388  usbTxLen1 = USBPID_NAK;
389 #endif
390  }else if(rq->bRequest == USBRQ_GET_INTERFACE){ /* 10 */
391  SET_REPLY_LEN(1);
392 #if USB_CFG_HAVE_INTRIN_ENDPOINT
393  }else if(rq->bRequest == USBRQ_SET_INTERFACE){ /* 11 */
394  USB_SET_DATATOKEN1(USBPID_DATA0); /* reset data toggling for interrupt endpoint */
395 # if USB_CFG_HAVE_INTRIN_ENDPOINT3
396  USB_SET_DATATOKEN3(USBPID_DATA0); /* reset data toggling for interrupt endpoint */
397 # endif
398 # if USB_CFG_IMPLEMENT_HALT
399  usbTxLen1 = USBPID_NAK;
400  }else if(rq->bRequest == USBRQ_CLEAR_FEATURE || rq->bRequest == USBRQ_SET_FEATURE){ /* 1|3 */
401  if(rq->wValue.bytes[0] == 0 && rq->wIndex.bytes[0] == 0x81){ /* feature 0 == HALT for endpoint == 1 */
402  usbTxLen1 = rq->bRequest == USBRQ_CLEAR_FEATURE ? USBPID_NAK : USBPID_STALL;
403  USB_SET_DATATOKEN1(USBPID_DATA0); /* reset data toggling for interrupt endpoint */
404 # if USB_CFG_HAVE_INTRIN_ENDPOINT3
405  USB_SET_DATATOKEN3(USBPID_DATA0); /* reset data toggling for interrupt endpoint */
406 # endif
407  }
408 # endif
409 #endif
410  }else{
411  /* the following requests can be ignored, send default reply */
412  /* 1: CLEAR_FEATURE, 3: SET_FEATURE, 7: SET_DESCRIPTOR */
413  /* 12: SYNCH_FRAME */
414  }
415  #undef SET_REPLY_LEN
416  }else{ /* not a standard request -- must be vendor or class request */
417  replyLen = usbFunctionSetup(data);
418  }
419 #if USB_CFG_IMPLEMENT_FN_READ || USB_CFG_IMPLEMENT_FN_WRITE
420  if(replyLen == 0xff){ /* use user-supplied read/write function */
421  if((rq->bmRequestType & USBRQ_DIR_MASK) == USBRQ_DIR_DEVICE_TO_HOST){
422  replyLen = rq->wLength.bytes[0]; /* IN transfers only */
423  }
424  flags &= ~USB_FLG_USE_DEFAULT_RW; /* we have no valid msg, use user supplied read/write functions */
425  }else /* The 'else' prevents that we limit a replyLen of 0xff to the maximum transfer len. */
426 #endif
427  if(!rq->wLength.bytes[1] && replyLen > rq->wLength.bytes[0]) /* limit length to max */
428  replyLen = rq->wLength.bytes[0];
429  }
430  /* make sure that data packets which are sent as ACK to an OUT transfer are always zero sized */
431  }else{ /* DATA packet from out request */
432 #if USB_CFG_IMPLEMENT_FN_WRITE
433  if(!(usbMsgFlags & USB_FLG_USE_DEFAULT_RW)){
434  uchar rval = usbFunctionWrite(data, len);
435  replyLen = 0xff;
436  if(rval == 0xff){ /* an error occurred */
437  usbMsgLen = 0xff; /* cancel potentially pending data packet for ACK */
438  usbTxLen = USBPID_STALL;
439  }else if(rval != 0){ /* This was the final package */
440  replyLen = 0; /* answer with a zero-sized data packet */
441  }
442  flags = 0; /* start with a DATA1 package, stay with user supplied write() function */
443  }
444 #endif
445  }
446  usbMsgFlags = flags;
447  usbMsgLen = replyLen;
448 }

452 static void usbBuildTxBlock(void)
453 {
454 uchar wantLen, len, txLen, token;
455 
456  wantLen = usbMsgLen;
457  if(wantLen > 8)
458  wantLen = 8;
459  usbMsgLen -= wantLen;
460  token = USBPID_DATA1;
461  if(usbMsgFlags & USB_FLG_TX_PACKET)
462  token = USBPID_DATA0;
463  usbMsgFlags++;
464  len = usbRead(usbTxBuf + 1, wantLen);
465  if(len <= 8){ /* valid data packet */
466  usbCrc16Append(&usbTxBuf[1], len);
467  txLen = len + 4; /* length including sync byte */
468  if(len < 8) /* a partial package identifies end of message */
469  usbMsgLen = 0xff;
470  }else{
471  txLen = USBPID_STALL; /* stall the endpoint */
472  usbMsgLen = 0xff;
473  }
474  usbTxBuf[0] = token;
475  usbTxLen = txLen;
476  DBG2(0x20, usbTxBuf, txLen-1);
477 }

493 USB_PUBLIC void usbPoll(void)
494 {
495 schar len;
496 uchar i;
497 
498  usbPollActive = 1 ; // mark usbPoll as active, used to stop overrun.
499 
500  if((len = usbRxLen) > 0){
501 /* We could check CRC16 here -- but ACK has already been sent anyway. If you
502  * need data integrity checks with this driver, check the CRC in your app
503  * code and report errors back to the host. Since the ACK was already sent,
504  * retries must be handled on application level.
505  * unsigned crc = usbCrc16(buffer + 1, usbRxLen - 3);
506  */
507  usbProcessRx(usbRxBuf + USB_BUFSIZE + 1 - usbInputBufOffset, len - 3);
508 #if USB_CFG_HAVE_FLOWCONTROL
509  if(usbRxLen > 0) /* only mark as available if not inactivated */
510  usbRxLen = 0;
511 #else
512  usbRxLen = 0; /* mark rx buffer as available */
513 #endif
514  }
515  if(usbTxLen & 0x10){ /* transmit system idle */
