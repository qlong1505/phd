200  EEDR = data[5];

208  EEAR = get_addr;
209  EEDR = data[4];

211  EECR |= 1 << EEMWE;
212  EECR |= 1 << EEWE;
213  sei();
214  }
215  }

217  if(get_addr == 0x31 || get_addr == 0x32)

219  else if(get_addr == 0x46 || get_addr == 0x48 || get_addr == 0x4A || get_addr == 0x4C || get_addr == 0x3E || get_addr == 0x24)

222  _MMIO_BYTE(get_addr) = data[5];
223  }
224  }

226  }

235  address.c[2] = *data++;

247  sei();

273  get_addr = (data[2] << 8) | data[3];

358  wdt_reset();

366  TCCR2 = 0x0B ; // Timer 2 as counter timer, count up to limit.
